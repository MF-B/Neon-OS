11/// Implementation of poll system call
12pub fn sys_poll(fds: UserPtr<pollfd>, nfds: usize, timeout_ms: c_int) -> LinuxResult<isize> {
13    debug!(
14        "sys_poll <= fds: {:?}, nfds: {}, timeout_ms: {}",
15        fds.address(),
16        nfds,
17        timeout_ms
18    );
19
20    if nfds == 0 {
21        if timeout_ms > 0 {
22            axtask::sleep(Duration::from_millis(timeout_ms as u64));
23        }
24        return Ok(0);
25    }
26
27    let deadline =
28        (!timeout_ms.is_negative()).then(|| wall_time() + Duration::from_millis(timeout_ms as u64));
29
30    loop {
31        axnet::poll_interfaces();
32
33        let mut ready_count = 0;
34        let pollfd_slice = fds.get_as_mut_slice(nfds)?;
35
36        for pollfd_data in pollfd_slice.iter_mut().take(nfds) {
37            pollfd_data.revents = 0;
38
39            if pollfd_data.fd < 0 {
40                continue;
41            }
42
43            match get_file_like(pollfd_data.fd) {
44                Ok(file) => match file.poll() {
45                    Ok(state) => {
46                        if (pollfd_data.events & POLLIN as i16) != 0 && state.readable {
47                            pollfd_data.revents |= POLLIN as i16;
48                        }
49                        if (pollfd_data.events & POLLOUT as i16) != 0 && state.writable {
50                            pollfd_data.revents |= POLLOUT as i16;
51                        }
52                        if pollfd_data.revents != 0 {
53                            ready_count += 1;
54                        }
55                    }
56                    Err(_) => {
57                        pollfd_data.revents = POLLERR as i16;
58                        ready_count += 1;
59                    }
60                },
61                Err(_) => {
62                    pollfd_data.revents = POLLNVAL as i16;
63                    ready_count += 1;
64                }
65            }
66        }
67
68        if ready_count > 0 {
69            return Ok(ready_count);
70        }
71
72        if deadline.is_some_and(|ddl| wall_time() >= ddl) {
73            return Ok(0);
74        }
75
76        axtask::sleep(Duration::from_millis(1));
77    }
78}

80/// Implementation of ppoll system call (simplified version without signal handling)
81pub fn sys_ppoll(
82    fds: UserPtr<pollfd>,
83    nfds: usize,
84    timeout: UserPtr<timespec>,
85    _sigmask: UserPtr<sigset_t>,
86) -> LinuxResult<isize> {
87    debug!("sys_ppoll <= fds: {:?}, nfds: {}", fds.address(), nfds);
88
89    if nfds == 0 {
90        if !timeout.is_null() {
91            let ts = timeout.get_as_mut()?;
92            let duration =
93                Duration::from_secs(ts.tv_sec as u64) + Duration::from_nanos(ts.tv_nsec as u64);
94            axtask::sleep(duration);
95        }
96        return Ok(0);
97    }
98
99    let deadline = if timeout.is_null() {
100        None
101    } else {
102        let ts = timeout.get_as_mut()?;
103        Some(
104            wall_time()
105                + Duration::from_secs(ts.tv_sec as u64)
106                + Duration::from_nanos(ts.tv_nsec as u64),
107        )
108    };
109
110    loop {
111        axnet::poll_interfaces();
112
113        let mut ready_count = 0;
114        let pollfd_slice = fds.get_as_mut_slice(nfds)?;
115
116        for pollfd_data in pollfd_slice.iter_mut().take(nfds) {
117            pollfd_data.revents = 0;

121            }

138                        ready_count += 1;

143                    ready_count += 1;
144                }
145            }
